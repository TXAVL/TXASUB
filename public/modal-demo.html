Basic Usage:

// Simple confirm const result = await txaModal.confirm('Bạn có muốn xóa item này?'); if (result) { console.log('User confirmed'); }

Custom Modal:

const result = await txaModal.show({ title: 'Custom Title', message: 'Custom message with subtitle', subtitle: 'This is a subtitle', type: 'info', icon: '🎯', confirmText: 'Custom OK', cancelText: 'Custom Cancel' });

Async Operation:

const result = await txaModal.confirmAsync( 'Xóa item này sẽ không thể hoàn tác?', 'Xác nhận xóa', async () => { // Simulate API call await new Promise(resolve => setTimeout(resolve, 2000)); return { success: true }; } );
